非常可乐

大家一定觉的运动以后喝可乐是一件很惬意的事情，但是seeyou却不这么认为。因为每次当seeyou买了可乐以后，阿牛就要求和seeyou一起分享这一瓶可乐，而且一定要喝的和seeyou一样多。但seeyou的手中只有两个杯子，它们的容量分别是N 毫升和M 毫升 可乐的体积为S （S<101）毫升　(正好装满一瓶) ，它们三个之间可以相互倒可乐 (都是没有刻度的，且 S==N+M，101＞S＞0，N＞0，M＞0) 。聪明的ACMER你们说他们能平分吗？如果能请输出倒可乐的最少的次数，如果不能输出"NO"。

Input

三个整数 : S 可乐的体积 , N 和 M是两个杯子的容量，以"0 0 0"结束。

Output

如果能平分的话请输出最少要倒的次数，否则输出"NO"。

Sample Input

7 4 3
4 1 3
0 0 0

Sample Output

NO
3

code

#include<queue>
#include<iostream>
using namespace std;
struct node{
    int deep;
    int s[3];
};
bool operator==(node a,node b){
    return a.s[0]==b.s[0]&&a.s[1]==b.s[1]&&a.s[2]==b.s[2];
}
int main(){
    int v[3];
    while(cin>>v[0]>>v[1]>>v[2]){
        if(v[0]==0&&v[1]==0&&v[2]==0)return 0;
        if(v[0]%2==1){
            cout<<"NO"<<endl;
            continue;
        }
        node temp[3];
        temp[0].s[0]=0;
        temp[0].s[1]=v[0]/2;
        temp[0].s[2]=v[0]/2;
        temp[1].s[0]=v[0]/2;
        temp[1].s[1]=0;
        temp[1].s[2]=v[0]/2;        
        temp[2].s[0]=v[0]/2;
        temp[2].s[1]=v[0]/2;
        temp[2].s[2]=0;
        
        node tmp;
        tmp.s[0]=v[0];
        tmp.s[1]=0;
        tmp.s[2]=0;
        tmp.deep=0;
        queue<node> q;
        q.push(tmp);
        int flag=0;
        bool Flag[100][100][100];
        for(int i=0;i<100;i++)
            for(int j=0;j<100;j++)
                for(int k=0;k<100;k++)
                    Flag[i][j][k]=false;
        Flag[tmp.s[0]][tmp.s[1]][tmp.s[2]]=true;
        while(!q.empty()){
            //cout<
            node tmp=q.front();
            if(tmp==temp[0]||tmp==temp[1]||tmp==temp[2])
            {
                flag=1;
                break;
            }
            q.pop();
            
            for(int i=0;i<3;i++)
                for(int j=0;j<3;j++){
                    if(i==j)continue;
                    if(tmp.s[i]!=v[i]&&tmp.s[j]!=0){
                        node tmp2;
                        tmp2.deep=tmp.deep+1;
                        tmp2.s[3-i-j]=tmp.s[3-i-j];
                        if(tmp.s[i]+tmp.s[j]>v[i]){
                            tmp2.s[i]=v[i];
                            tmp2.s[j]=tmp.s[i]+tmp.s[j]-v[i];
                        }
                        else{
                            tmp2.s[i]=tmp.s[i]+tmp.s[j];
                            tmp2.s[j]=0;    
                        }
                        if(!Flag[tmp2.s[0]][tmp2.s[1]][tmp2.s[2]]){
                            Flag[tmp2.s[0]][tmp2.s[1]][tmp2.s[2]]=true;
                            q.push(tmp2);
                        }
                    }
                }
        }
        if(flag==0){
            cout<<"NO"<<endl;
        }
        else{
            cout<<q.front().deep<<endl;
        }
        
        
    }
}
