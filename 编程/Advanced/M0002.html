Largest prime factor

Everybody knows any number can be combined by the prime number.
Now, your task is telling me what position of the largest prime factor.
The position of prime 2 is 1, prime 3 is 2, and prime 5 is 3, etc.
Specially, LPF(1) = 0.

Input

Each line will contain one integer n(0 < n < 1000000).

Output

Output the LPF(n).

Sample Input

1
2
3
4
5

Sample Output

0
1
2
1
3

code

//数据量大必须用scanf & printf
//动态规划，素数筛选法
#include<iostream>
#include<cstdio>
using namespace std;
bool Ero[1000000];
int position[1000000];
int result(int n) {
    if (position[n] != -1)return position[n];
    int r = -1;
    if (n % 2 == 0)r = result(n / 2);
    else {
        for (int i = 3; i*i <= n; i += 2)
            if (n%i == 0) {
                r = result(n / i);
                break;
            }
    }
    position[n] = r;
    return r;
}
int main() {
    for (int i = 0; i<1000000; i++)
        Ero[i] = true;
    for (int i = 2; i<1000; i++) {
        if (!Ero[i])continue;
        for (int j = i; i*j < 1000000; j++)
            Ero[i*j] = false;
    }
    position[1] = 0;
    for (int i = 2, count = 1; i<1000000; i++) {
        if (Ero[i]) {
            position[i] = count;
            count++;
        }
        else {
            position[i] = -1;
            position[i] = result(i);
        }
    }
    int n;
    while (scanf("%d",&n)!=EOF) {
        printf("%d\n",position[n]);
    }
}
