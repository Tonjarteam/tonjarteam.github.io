Number Sequence

A number sequence is defined as follows:
f(1) = 1, f(2) = 1, f(n) = (A * f(n - 1) + B * f(n - 2)) mod 7.
Given A, B, and n, you are to calculate the value of f(n).

Input

The input consists of multiple test cases. Each test case contains 3 integers A, B and n on a single line (1 <= A, B <= 1000, 1 <= n <= 100,000,000). Three zeros signal the end of input and this test case is not to be processed.

Output

For each test case, print the value of f(n) on a single line.

Sample Input

1 1 3
1 2 10
0 0 0

Sample Output

2
5

code

#include<iostream>
using namespace std;
struct Mat{
    unsigned long long a,b,c,d;
};
void setMat(Mat& M,unsigned long long a,unsigned long long b,unsigned long long c,unsigned long long d){
    M.a=a;
    M.b=b;
    M.c=c;
    M.d=d;
}
Mat operator*(Mat a,Mat b){
    setMat(a,(a.a*b.a+a.b*b.c)%7,(a.a*b.b+a.b*b.d)%7,(a.c*b.a+a.d*b.c)%7,(a.c*b.b+a.d*b.d)%7);
    return a;
}
Mat qiumi(Mat M,int n){
    if(n==1)return M;
    Mat a=qiumi(M,n/2);
    if(n%2==0)return a*a;
    return a*a*M;
}
int main(){
    int a,b,n;
    while(cin>>a>>b>>n){
        if(a==0&&b==0&&n==0)return 0;
        if(n<3){
            cout<<1<<endl;
            continue;
        }
        a%=7;
        b%=7;
        Mat M;
        setMat(M,a,b,1,0);
        M=qiumi(M,n-2);
        cout<<(M.a+M.b)%7<<endl;
    }
}
